import plotly.graph_objects as go

# initialize figure
fig = go.Figure()


# Empty Lists
sliders = []
buttonList = []

# Creating an empty visibility list of False values that we will fill
numYears = len(grouped_df.year.unique())
numStates = len(grouped_df.STNAME.unique())

# Add traces, one for each slider step

    steps = []

    # Filtered state dataframe
    stateDF = grouped_df[grouped_df.STNAME == state]

    # Displaying the first year for each state
    state_first_visibility = visibility_list.copy()
    state_first_visibility[counter] = True

    # For loop by year
    for year in stateDF.year.unique():
        # Each year has its own visibility list
        visibility_list1 = visibility_list.copy()
        visibility_list1[counter] = True
        # Adding to the counter
        counter += 1

        # Steps on the slider
        step = {
            "method": "update",
            "args": [{"visible": visibility_list1}],
            "label": str(year),
        }
        steps.append(step)

        # ID
        id = str(year) + str(state)

        # Dataframe of just this year
        dateDF = stateDF[stateDF.year == year]

        # Adding the tace
        fig.add_trace(
            go.Scatter(
                uid=id,
                visible=False,
                line=dict(color="green", width=6),
                name="Year: " + str(year),
                x=dateDF.date,
                y=dateDF.ASSET,
            )
        )

    # Creating a slider for the year
    slider1 = [
        dict(active=0, currentvalue={"prefix": "Year: "}, pad={"t": 50}, steps=steps)
    ]
    sliders.append(slider1)

    # Making the button
    tmpbutton = dict(
        label=str(state),
        method="update",
        args=[{"visible": state_first_visibility}, {"sliders": slider1}],
    )

    # Adding the button
    buttonList.append(tmpbutton)

fig.data[0].visible = True

import plotly.graph_objects as go
import plotly.express as px

# initialize figure

# Empty Lists

# Filtering to just the states we want to focus on
states = ["SOUTH DAKOTA", "NEW YORK", "CALIFORNIA", "NORTH CAROLINA", "OHIO"]

df2 = df2[[True if state in states else False for state in df2["STNAME"]]]

# Creating an empty visibility list of False values that we will fill
numDecade = len(df2.decade.unique())
numStates = len(df2.STNAME.unique())
visibility_list = [False] * numStates

# Counter for instances
counter = 0

# Add traces, one for each slider step
for i, state in enumerate(df2.STNAME.unique()):
    steps = []

    # Filtered state dataframe
    stateDF = df2[df2.STNAME == state]

    # Displaying the first year for each state
    state_first_visibility = visibility_list.copy()
    state_first_visibility[counter] = True
    state_df.sort_values(by="date", ascending=True, inplace=True)
    
    
    # Each  has its own visibility list
    visibility_list1 = visibility_list.copy()
    visibility_list1[counter] = True
    # Adding to the counter
    counter += 1

    # Steps on the slider
    step = {
        "method": "update",
        "args": [{"visible": visibility_list1}],
        "label": str(decade),
    }
    steps.append(step)

    # ID
    id = str(decade) + str(state)

    # Dataframe of just this decade
    dateDF = stateDF[stateDF.decade == decade]

    # Making the bank names prettier

    for bank in stateDF["top_bank"]:
        if bank == "JPMORGAN CHASE BANK NA":
            name = "JP Morgan Chase"
        elif bank == "BANK OF AMERICA NA":
            name = "Bank of America"
        elif bank == "WELLS FARGO BANK NA":
            name = 'Wells Fargo'
        elif bank == "CITIBANK NATIONAL ASSN":
            name = "Citibank"
        else:
            name = 'All Other Banks'
